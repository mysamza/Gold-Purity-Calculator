$(function () {
$('#btn').click(function () {
$('#resulttext').hide();
var ihaveweight = $('#ihaveweight').val();
var ihavepurity = $('#ihavepurity').val();
var iwantweight = $('#iwantweight').val();
var iwantpurity = $('#iwantpurity').val();
var unit = $('#unit').val();
var result = 0;
if (ihavepurity > iwantpurity) {
result = (((ihaveweight * ihavepurity / 1000) / iwantpurity) * 1000) - ihaveweight;
var addedmetal = (parseFloat(ihaveweight) + parseFloat(result)).toFixed(3);
$('#resulttext').html('Add ' + result.toFixed(3).bold() + ' ' + unit.bold() + ' of copper to your ' + ihaveweight + ' ' + unit + '.' + ' Your final metal will be ' + addedmetal + ' ' + unit).show();
} else {
ihavepurity = 1000 - ihavepurity
iwantpurity = 1000 - iwantpurity
result = (((ihaveweight * ihavepurity / 1000) / iwantpurity) * 1000) - ihaveweight
iwantpurity = -iwantpurity + 1000;
var addedmetal = (parseFloat(ihaveweight) + parseFloat(result)).toFixed(3);
$('#resulttext').html('Add ' + result.toFixed(3).bold() + ' ' + unit.bold() + ' of pure gold (99.99%) to your ' + ihaveweight + ' ' + unit + '.' + ' Your final metal will be ' + addedmetal + ' ' + unit).show();
}
});
});
